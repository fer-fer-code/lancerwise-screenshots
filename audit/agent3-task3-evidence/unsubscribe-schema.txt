=== 20260425000001_email_unsubscribe_recurring.sql ===
-- Add email_unsubscribed to profiles
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS email_unsubscribed boolean DEFAULT false;

-- Recurring invoice schedules
CREATE TABLE IF NOT EXISTS recurring_schedules (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users NOT NULL,
  client_id uuid REFERENCES clients(id) ON DELETE SET NULL,
  client_name text NOT NULL,
  service_description text NOT NULL,
  amount numeric(12,2) NOT NULL DEFAULT 0,
  currency text NOT NULL DEFAULT 'USD',
  frequency text NOT NULL DEFAULT 'monthly', -- weekly | biweekly | monthly | quarterly
  next_date date NOT NULL,
  active boolean DEFAULT true,
  invoices_created integer DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

ALTER TABLE recurring_schedules ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users manage their recurring schedules" ON recurring_schedules
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

=== 20260512000001_clients_email_unsubscribed.sql ===
-- D1d B1 — client-side email opt-out for outbound-to-client recurring emails.
--
-- Previously only profiles.email_unsubscribed existed (for freelancer/user
-- self-notifications). Outbound-to-client cron emails (invoice reminders,
-- NPS surveys, special-date greetings, etc.) had no clean way to honor
-- a per-client opt-out. CAN-SPAM requires a working unsubscribe mechanism
-- for recurring commercial messages — this migration provides it.

-- 1. Per-client opt-out flag.
ALTER TABLE clients ADD COLUMN IF NOT EXISTS email_unsubscribed boolean NOT NULL DEFAULT false;

-- Partial index — most clients are NOT unsubscribed; the index lets
-- "WHERE email_unsubscribed = true" / "WHERE email_unsubscribed = false"
-- pre-filters stay cheap as the table grows.
CREATE INDEX IF NOT EXISTS clients_email_unsubscribed_idx
  ON clients(email_unsubscribed)
  WHERE email_unsubscribed = true;

-- 2. Audit log of unsubscribe events. Replaces the implicit best-effort
-- table the legacy /api/unsubscribe route tried to insert into. Columns
-- map directly onto the verified HMAC payload (audience + target_id + email)
-- so we can correlate clicks to specific recipients without leaking PII
-- through the URL.
CREATE TABLE IF NOT EXISTS email_unsubscribe_log (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  audience text NOT NULL CHECK (audience IN ('user', 'client')),
  target_id uuid NOT NULL,
  email text NOT NULL,
  source text NOT NULL DEFAULT 'unsubscribe_link',
  unsubscribed_at timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS email_unsubscribe_log_target_idx
  ON email_unsubscribe_log(audience, target_id);
CREATE INDEX IF NOT EXISTS email_unsubscribe_log_unsubscribed_at_idx
  ON email_unsubscribe_log(unsubscribed_at DESC);

-- RLS: only the service role writes here (via the admin Supabase client
-- from the public unsubscribe endpoint). No public SELECT — clients and
-- users have no reason to see arbitrary unsubscribe events. Enable RLS
-- without a SELECT policy → effectively service-role-only access.
ALTER TABLE email_unsubscribe_log ENABLE ROW LEVEL SECURITY;

=== 20260425000039_email_logs.sql (excerpt) ===
CREATE TABLE IF NOT EXISTS email_logs (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES profiles(id) ON DELETE CASCADE,
  type text NOT NULL,
  to_email text NOT NULL,
  subject text NOT NULL,
  sent_at timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS email_logs_user_id_sent_at ON email_logs(user_id, sent_at DESC);

ALTER TABLE email_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view their own email logs"
  ON email_logs FOR SELECT
  USING (auth.uid() = user_id);
